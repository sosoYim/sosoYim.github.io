### 내용 정리

자바스크립트로 자식창에서 부모창 제어하기

```javascript
//부모창에서================================================
//자식창에서 찾아올 값을 가진 input
<input type="hidden" id="p_value" value="hi my child">

//자식창에서 호출할 함수
function p_function(msg){
	alert(msg);    
}

//자식창에서================================================
//부모창 값 가져오기 : hi my child
let p_value = window.opener.document.getElementById("p_value").value;
//부모창 함수 호출하기
opener.p_functin(p_value);
//=========================================================

//부모창 새로고침
opener.location.reload();

//+ 함께 자주 쓰는 기능 : 자기 자신 닫기
window.open('채팅방이름(자식)').close();

//주의!!!! 자식창과 부모창들 사이에서 제어하려면, 팝업으로 띄워진 자식창에 이름이 반드시 부여되어야한다! 아래의 이름 부분에 값을 꼭 넣어주자. 이거 빈 값 넣고 부모창 제어하려다 많은 시간을 허비했다...
//window.open('경로','이름','설정');

```
